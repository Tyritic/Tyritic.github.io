class Solution {
    public boolean isAnagram(String s, String t) {
        int[] record = new int[26];
        for(int i=0;i<s.length();i++)
        {
            record[s.charAt(i)-'a']++;
        }
        for(int i=0;i<t.length();i++)
        {
            record[t.charAt(i)-'a']--;
        }
        for(int count:record)
        {
            if(count!=0)
                return false;
        }
        return true;

    }
}

        // ‰øùÂ≠òsumÁöÑÂá∫Áé∞ÊÉÖÂÜµ

class Solution {
    public List<List<String>> groupAnagrams(String[] strs) {
        HashMap<String,List<String>>map=new HashMap<String, List<String>>();
        for(String str:strs)
        {
            // Â∞ÜÊØè‰∏™Â≠óÁ¨¶‰∏≤ÊéíÂ∫è

    public int[] intersect(int[] nums1, int[] nums2) {
        HashMap<Integer,Integer>map=new HashMap<>();
        int[]ans=new int[nums1.length];
        // ËÆ∞ÂΩïnums1‰∏≠ÁöÑÊï∞Â≠óÂá∫Áé∞È¢ëÁéá
        for(Integer num1:nums1)
        {

        // ËÆ∞ÂΩï‰∫§ÈõÜÁöÑÈïøÂ∫¶
        int index=0;
        // ÈÅçÂéÜnums2
        for(Integer num2:nums2)

            int count=map.getOrDefault(num2,0);

                // Â∞Ünums2‰∏≠ÁöÑÊï∞Â≠óÂä†ÂÖ•Á≠îÊ°àÈõÜ
                ans[index++]=num2;

                count--;
                // Êõ¥Êñ∞map
                if(count>0)

                    map.put(num2,count);
                }
                else {
                    map.remove(num2);
                }
            }
        }
        return Arrays.copyOf(ans,index);
    }

class Solution {
    public int[] twoSum(int[] nums, int target) {
        // ‰ΩøÁî®mapÈõÜÂêàËÆ∞ÂΩïÂâçÈù¢ÁöÑÊï∞ÁªÑÂÖÉÁ¥†
        HashMap<Integer,Integer>map=new HashMap<>();

            return res;
        }
        for(int i=0;i<nums.length;i++)
        {
            // ÂØªÊâætempÊòØÂê¶Âú®ÂìàÂ∏åË°®‰∏≠
            int temp=target-nums[i];
            if(map.containsKey(temp))

                break;
            }
            map.put(nums[i],i);
        }
        return res;
    }
}

class Solution {
    public int fourSumCount(int[] nums1, int[] nums2, int[] nums3, int[] nums4) {
        int res=0;
        HashMap<Integer,Integer>map=new HashMap<>();
        // ÁªüËÆ°nums1Âíånums2ÁöÑÂÖÉÁ¥†‰πãÂíå‰ª•ÂèäÂá∫Áé∞Ê¨°Êï∞
        for(Integer i:nums1)
        {
            for(Integer j:nums2)
            {
                int sum=i+j;
                map.put(sum,map.getOrDefault(sum,0)+1);
            }
        }
        // ÁªüËÆ°nums3Âíånums4ÁöÑÂÖÉÁ¥†‰πãÂíåÁöÑÁõ∏ÂèçÊï∞ÊòØÂê¶Âú®map‰∏≠Âá∫Áé∞
        for(Integer i:nums3)
        {
            for(Integer j:nums4)
            {
                res+=map.getOrDefault(0-i-j,0);
            }
        }
        return res;
    }
}
