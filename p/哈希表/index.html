class Solution {
    public boolean isAnagram(String s, String t) {
        int[] record = new int[26];
        for(int i=0;i<s.length();i++)
        {
            record[s.charAt(i)-'a']++;
        }
        for(int i=0;i<t.length();i++)
        {
            record[t.charAt(i)-'a']--;
        }
        for(int count:record)
        {
            if(count!=0)
                return false;
        }
        return true;

    }

class Solution {
    public int[] intersect(int[] nums1, int[] nums2) {
        HashMap<Integer,Integer>map=new HashMap<>();
        int[]ans=new int[nums1.length];
        // è®°å½•nums1ä¸­çš„æ•°å­—å‡ºç°é¢‘ç‡
        for(Integer num1:nums1)
        {
            map.put(num1, map.getOrDefault(num1,0)+1);
        }
        // è®°å½•äº¤é›†çš„é•¿åº¦
        int index=0;
        // éå†nums2

        {

                ans[index++]=num2;
                // nums2ä¸­çš„æ•°å­—åœ¨nums1ä¸­çš„å‡ºç°æ¬¡æ•°å‡ä¸€
                count--;
                // æ›´æ–°map
                if(count>0)

        return Arrays.copyOf(ans,index);

    public int[] twoSum(int[] nums, int target) {
        // ä½¿ç”¨mapé›†åˆè®°å½•å‰é¢çš„æ•°ç»„å…ƒç´ 
        HashMap<Integer,Integer>map=new HashMap<>();
        int[]res=new int[2];

            return res;
        }
        for(int i=0;i<nums.length;i++)
        {
            // å¯»æ‰¾tempæ˜¯å¦åœ¨å“ˆå¸Œè¡¨ä¸­
            int temp=target-nums[i];
            if(map.containsKey(temp))

        }
        return res;
    }

class Solution {
    public int fourSumCount(int[] nums1, int[] nums2, int[] nums3, int[] nums4) {
        int res=0;
        HashMap<Integer,Integer>map=new HashMap<>();
        // ç»Ÿè®¡nums1å’Œnums2çš„å…ƒç´ ä¹‹å’Œä»¥åŠå‡ºç°æ¬¡æ•°
        for(Integer i:nums1)
        {
            for(Integer j:nums2)
            {
                int sum=i+j;
                map.put(sum,map.getOrDefault(sum,0)+1);
            }
        }
        // ç»Ÿè®¡nums3å’Œnums4çš„å…ƒç´ ä¹‹å’Œçš„ç›¸åæ•°æ˜¯å¦åœ¨mapä¸­å‡ºç°
        for(Integer i:nums3)
        {
            for(Integer j:nums4)
            {
                res+=map.getOrDefault(0-i-j,0);
            }
        }
        return res;
    }
}
