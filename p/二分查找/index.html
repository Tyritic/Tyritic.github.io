class Solution {
    public int[] searchRange(int[] nums, int target) {
        int index=BinarySearch(nums,target); // 1.å…ˆè¿›è¡ŒäºŒåˆ†æŸ¥æ‰¾

    public static int BinarySearch(int[] nums, int target) {
        int left=0;
        int right=nums.length-1;

        {
            int mid=left+(right-left)/2;
            if(nums[mid]<target)
            {
                left=mid+1;
            }
            else if(nums[mid]>target)
            {
                right=mid-1;
            }
            else{

            }
        }

class Solution {
    public boolean isPerfectSquare(int num) {
        int left=0;
        int right=num;
        if(num==0||num==1)return true;// 1.ç‰¹æ®Šå€¼åˆ¤æ–­
        while(left<=right)
        {
            int mid=left+(right-left)/2;
            if(mid==num/(mid*1.0))
                return true;
            else if (mid>num/(mid*1.0))
                right=mid-1;
            else 
                left=mid+1;
        }
        return false;
    }
