class Solution {
    public int totalFruit(int[] fruits) {
        int n=fruits.length;
        if(n==1)return 1;
        if(n==2)return 2;
        int left=0;
        int ans=0;
        int result=1;
        HashMap<Integer,Integer>map=new HashMap<>();
        for(int right=0;right<fruits.length;right++)
        {
            // 1.ËÆ°ÁÆóÁ∫¶ÊùüÊù°‰ª∂ÔºàÊ∞¥ÊûúÁöÑÁßçÁ±ªÊï∞Ôºâ
            map.put(fruits[right],map.getOrDefault(fruits[right],0)+1);
            // 2.Ëã•‰∏çÊª°Ë∂≥Êù°‰ª∂
            while(map.size()>2)
            {
                // 3.Âè≥ÁßªÂ∑¶ËæπÁïå
                map.put(fruits[left],map.get(fruits[left])-1);
                if(map.get(fruits[left])==0)
                    map.remove(fruits[left]);
                left++;
            }
            // 4.Êõ¥Êñ∞ÁªìÊûú
           result=Math.max(result,right-left+1);
        }
        return result;
    }
}

class Solution {
    public String minWindow(String s, String t) {
       	int[]cnS=new int[128];
       	int[]cnT=new int[128];
       	char[]S=s.toCharArray();
       	char[]T=t.toCharArray();
       	for(char e:T)
       	{
           cnT[e]++;
       	}
       	int left=0;
 		int ansLeft=-1;
        int ansRight=s.length-1;
        for(int right=0;right<s.length;right++)
        {
            // 1.ËÆ°ÁÆóÁ∫¶ÊùüÊù°‰ª∂
            cnS[S[right]]++;
            // 2.Ëã•Êª°Ë∂≥Êù°‰ª∂
            while(isCover(cnS,cnT))
            {
                // 3.Êõ¥Êñ∞ÁªìÊûú
                if(right-left+1<ansRight-ansLeft+1)
                {
                    ansLeft=left;
                    ansRight=right;
                }
                // 4.Âè≥ÁßªÂ∑¶ËæπÁïå
                cnT[S[left]]--;
                left++;
            }
        }
        return ansLeft<0?"":s.substring(ansLeft,ansRight+1);
    }
    
    //ÈÄöËøáÁªüËÆ°Â≠ê‰∏≤‰∏≠Â≠óÁ¨¶ÁöÑÂá∫Áé∞Ê¨°Êï∞Êù•Âà§Êñ≠ÊòØÂê¶Ë¶ÜÁõñ
    public static boolean isCover(int[]cnS,int[]cnT)
    {
        for(int i='a';i<='z';i++)
        {
            if(cnS[i]<cnT[i])
            {
                return false;
            }
        }
        for(int i='A';i<='Z';i++)
        {
            if(cnS[i]<cnT[i])
            {
                return false;
            }
        }
        return true;
    }
}
