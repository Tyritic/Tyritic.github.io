    public int[] maxSlidingWindow(int[] nums, int k) {
        if (nums.length == 1) {

class MyQueue{
    Deque<Integer>deq=new LinkedList<>();
    void poll(int val){
        if(!deq.isEmpty()&&val==deq.peek())
            deq.poll();
    }
    void push(int val)
    {
        while (!deq.isEmpty() && val > deq.getLast()) {
            deq.removeLast();
        }
        deq.add(val);
    }
    int peek()
    {
        return deq.peek();
    }
}
